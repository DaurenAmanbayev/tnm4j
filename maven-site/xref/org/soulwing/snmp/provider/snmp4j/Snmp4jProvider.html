18  package org.soulwing.snmp.provider.snmp4j;
19  
20  import static org.soulwing.snmp.provider.snmp4j.Snmp4jLogger.logger;
21  
22  import java.io.IOException;
23  import java.util.IdentityHashMap;
24  import java.util.concurrent.locks.Lock;
25  import java.util.concurrent.locks.ReentrantLock;
26  
27  import org.snmp4j.Snmp;
28  import org.snmp4j.Target;
29  import org.snmp4j.smi.Address;
30  import org.snmp4j.smi.GenericAddress;
31  import org.snmp4j.smi.SMIAddress;
32  import org.snmp4j.smi.SMIConstants;
33  import org.snmp4j.smi.UdpAddress;
34  import org.snmp4j.transport.DefaultUdpTransportMapping;
35  import org.soulwing.snmp.Mib;
36  import org.soulwing.snmp.SnmpContext;
37  import org.soulwing.snmp.SnmpException;
38  import org.soulwing.snmp.SnmpListener;
39  import org.soulwing.snmp.SnmpTarget;
40  import org.soulwing.snmp.SnmpTargetConfig;
41  import org.soulwing.snmp.provider.SnmpProvider;
42  
43  /**
44   * An {@link SnmpProvider} based on SNMP4j.
45   *
46   * @author Carl Harris
47   */
48  public class Snmp4jProvider implements SnmpProvider, DisposeListener {
49  
50    private static final String PROVIDER_NAME = "snmp4j";
51  
52    private static final TargetStrategy[] targetStrategies = {
53        new CommunityTargetStrategy(),
54        new UserTargetStrategy()
55    };
56  
57    private static final PduFactoryStrategy[] pduFactoryStrategies = {
58        new SnmpV2cPduFactoryStrategy(),
59        new SnmpV3PduFactoryStrategy()
60    };
61  
62    private final Lock lock = new ReentrantLock();
63  
64    private final IdentityHashMap<Object, Object> refs =
65        new IdentityHashMap<Object, Object>();
66  
67    private volatile Snmp snmp;
68  
69    /**
70     * {@inheritDoc}
71     */
72    @Override
73    public String getName() {
74      return PROVIDER_NAME;
75    }
76  
77    /**
78     * {@inheritDoc}
79     */
80    @Override
81    public SnmpContext newContext(SnmpTarget target, SnmpTargetConfig config, 
82        Mib mib) {
83      lock.lock();
84      try {
85        Target snmp4jTarget = createTarget(target);
86        PduFactory pduFactory = createPduFactory(target);
87        snmp4jTarget.setAddress(createAddress(target));
88        snmp4jTarget.setRetries(config.getRetries());
89        snmp4jTarget.setTimeout(config.getTimeout());
90        Snmp4jContext context = new Snmp4jContext(target, config, mib,
91            getSnmp(), snmp4jTarget, pduFactory,
92            new SimpleVarbindFactory(mib), this);
93        refs.put(context, context);
94        return context;
95      }
96      finally {
97        lock.unlock();
98      }
99    }
100 
101   @Override
102   public SnmpListener newListener(String address, int port, Mib mib) {
103     lock.lock();
104     try {
105       Address listenAddress = address == null ?
106           new UdpAddress(port) : createAddress(address, port);
107       Snmp4jListener listener = new Snmp4jListener(getSnmp(),
108           listenAddress, new Snmp4jNotificationEventFactory(
109           new SimpleVarbindFactory(mib)), this);
110       listener.open();
111       refs.put(listener, listener);
112       return listener;
113     }
114     finally {
115       lock.unlock();
116     }
117   }
118 
119   private Snmp getSnmp() {
120     if (snmp == null) {
121       try {
122         logger.info("starting SNMP listener");
123         snmp = new Snmp(new DefaultUdpTransportMapping());
124         snmp.listen();
125       }
126       catch (IOException ex) {
127         throw new SnmpException("error creating SNMP session", ex);
128       }
129     }
130     return snmp;
131   }
132 
133   @Override
134   public void close() {
135     lock.lock();
136     try {
137       refs.clear();
138       shutdown();
139     }
140     finally {
141       lock.unlock();
142     }
143   }
144 
145   @Override
146   public void onDispose(Object ref) {
147     lock.lock();
148     try {
149       refs.remove(ref);
150       if (!refs.isEmpty()) return;
151       shutdown();
152     }
153     finally {
154       lock.unlock();
155     }
156   }
157 
158   private void shutdown() {
159     try {
160       snmp.close();
161     }
162     catch (IOException ex) {
163       Snmp4jLogger.logger.warn("while closing SNMP session: {}", ex.toString(), ex);
164     }
165   }
166 
167   private static Target createTarget(SnmpTarget target) {
168     for (TargetStrategy strategy : targetStrategies) {
169       Target snmp4jTarget = strategy.newTarget(target);
170       if (snmp4jTarget != null) return snmp4jTarget;
171     }
172     throw new RuntimeException("unsupported target type");
173   }
174 
175   private static Address createAddress(SnmpTarget target) {
176     return createAddress(target.getAddress(), target.getPort());
177   }
178 
179   private static Address createAddress(String address, int port) {
180     Assert.notNull(address, "address is required");
181     StringBuilder sb = new StringBuilder();
182     sb.append("udp:");
183     sb.append(address);
184     sb.append("/").append(port);
185     return GenericAddress.parse(sb.toString());
186   }
187 
188   private static PduFactory createPduFactory(SnmpTarget target) {
189     for (PduFactoryStrategy strategy : pduFactoryStrategies) {
190       PduFactory factory = strategy.newPduFactory(target);
191       if (factory != null) return factory;
192     }
193     throw new RuntimeException("unsupported target type");
194   }
195

197 }
