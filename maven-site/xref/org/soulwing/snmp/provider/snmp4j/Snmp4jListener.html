19  package org.soulwing.snmp.provider.snmp4j;
20  
21  import static org.soulwing.snmp.provider.snmp4j.Snmp4jLogger.logger;
22  
23  import java.util.ArrayList;
24  import java.util.List;
25  import java.util.concurrent.locks.ReadWriteLock;
26  import java.util.concurrent.locks.ReentrantReadWriteLock;
27  
28  import org.snmp4j.CommandResponder;
29  import org.snmp4j.CommandResponderEvent;
30  import org.snmp4j.Snmp;
31  import org.snmp4j.smi.Address;
32  import org.soulwing.snmp.SnmpException;
33  import org.soulwing.snmp.SnmpListener;
34  import org.soulwing.snmp.SnmpNotificationEvent;
35  import org.soulwing.snmp.SnmpNotificationHandler;
36  
37  /**
38   * An {@link SnmpListener} implementation based on SNMP4j.
39   *
40   * @author Carl Harris
41   */
42  class Snmp4jListener implements SnmpListener, CommandResponder {
43  
44    private final ReadWriteLock lock = new ReentrantReadWriteLock();
45  
46    private final List<PrioritizedHandler> handlers =
47        new ArrayList<PrioritizedHandler>();
48  
49    private final Snmp snmp;
50    private final Address listenAddress;
51    private final SnmpNotificationEventFactory eventFactory;
52    private final DisposeListener disposeListener;
53  
54    private volatile boolean closed;
55  
56    public Snmp4jListener(Snmp snmp, Address listenAddress,
57        SnmpNotificationEventFactory eventFactory,
58        DisposeListener disposeListener) {
59      this.snmp = snmp;
60      this.listenAddress = listenAddress;
61      this.eventFactory = eventFactory;
62      this.disposeListener = disposeListener;
63    }
64  
65    @Override
66    public void addHandler(SnmpNotificationHandler handler) {
67      addHandler(handler, Integer.MAX_VALUE);
68    }
69  
70    @Override
71    public void addHandler(SnmpNotificationHandler handler, int priority) {
72      assertNotClosed();
73      PrioritizedHandler wrapper = new PrioritizedHandler(priority, handler);
74      if (handlers.contains(wrapper)) return;
75      lock.writeLock().lock();
76      try {
77        int index = priority < Integer.MAX_VALUE ? 0 : handlers.size();
78        while (index < handlers.size()
79            && handlers.get(index).priority <= priority) {
80          index++;
81        }
82        handlers.add(index, wrapper);
83      }
84      finally {
85        lock.writeLock().unlock();
86      }
87    }
88  
89    @Override
90    public void removeHandler(SnmpNotificationHandler handler) {
91      assertNotClosed();
92      lock.writeLock().lock();
93      try {
94        handlers.remove(new PrioritizedHandler(0, handler));
95      }
96      finally {
97        lock.writeLock().unlock();
98      }
99    }
100 
101   private void assertNotClosed() {
102     if (!closed) return;
103     throw new IllegalStateException("listener is closed");
104   }
105 
106   public void open() throws SnmpException {
107     if (!snmp.addNotificationListener(listenAddress, this)) {
108       throw new SnmpException("failed to create listener");
109     }
110     logger.info("listening for notifications at {}", listenAddress);
111   }
112 
113   @Override
114   public void close() {
115     if (closed) return;
116     lock.writeLock().lock();
117     try {
118       snmp.removeNotificationListener(listenAddress);
119       handlers.clear();
120       closed = true;
121       logger.info("closed listener at {}", listenAddress);
122       disposeListener.onDispose(this);
123     }
124     finally {
125       lock.writeLock().unlock();
126     }
127   }
128 
129   @Override
130   public void processPdu(CommandResponderEvent ev) {
131     if (closed) return;
132     if (logger.isTraceEnabled()) {
133       logger.trace("listener {} received event: {}", this, ev);
134     }
135     try {
136       notifyHandlers(eventFactory.newEvent(this, ev));
137     }
138     catch (RuntimeException ex) {
139       logger.warn("listener exception: {}", ex.toString(), ex);
140     }
141   }
142 
143   void notifyHandlers(SnmpNotificationEvent event) {
144     lock.readLock().lock();
145     try {
146       if (logger.isDebugEnabled()) {
147         logger.debug("listener {} notifying handlers for event: {}",
148             this, event);
149       }
150       for (PrioritizedHandler wrapper : handlers) {
151         // we use a wrapper type here to allow for scripting languages that
152         // don't require methods to return a value
153         Boolean handled = wrapper.handler.handleNotification(event);
154         if (handled != null && handled) {
155           if (logger.isDebugEnabled()) {
156             logger.debug("event was handled by {}", wrapper.handler);
157           }
158           break;
159         }
160       }
161     }
162     finally {
163       lock.readLock().unlock();
164     }
165   }
166 
167   @Override
168   public String toString() {
169     return listenAddress.toString();
170   }
171 
172   static class PrioritizedHandler {
173     final int priority;
174     final SnmpNotificationHandler handler;
175 
176     PrioritizedHandler(int priority, SnmpNotificationHandler handler) {
177       this.priority = priority;
178       this.handler = handler;
179     }
180 
181     @Override
182     public boolean equals(Object obj) {
183       if (obj == this) return true;
184       if (!(obj instanceof PrioritizedHandler)) return false;
185       return this.handler.equals(((PrioritizedHandler) obj).handler);
186     }
187 
188     @Override
189     public int hashCode() {
190       return handler.hashCode();
191     }
192   }
