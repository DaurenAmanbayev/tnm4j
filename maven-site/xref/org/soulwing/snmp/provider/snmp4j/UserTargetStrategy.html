18  package org.soulwing.snmp.provider.snmp4j;
19  
20  import org.snmp4j.Target;
21  import org.snmp4j.UserTarget;
22  import org.snmp4j.mp.MPv3;
23  import org.snmp4j.mp.SnmpConstants;
24  import org.snmp4j.security.AuthMD5;
25  import org.snmp4j.security.AuthSHA;
26  import org.snmp4j.security.Priv3DES;
27  import org.snmp4j.security.PrivAES128;
28  import org.snmp4j.security.PrivAES192;
29  import org.snmp4j.security.PrivAES256;
30  import org.snmp4j.security.PrivDES;
31  import org.snmp4j.security.SecurityLevel;
32  import org.snmp4j.security.SecurityProtocols;
33  import org.snmp4j.security.USM;
34  import org.snmp4j.security.UsmUser;
35  import org.snmp4j.smi.OID;
36  import org.snmp4j.smi.OctetString;
37  import org.soulwing.snmp.SnmpTarget;
38  import org.soulwing.snmp.SnmpV3Target;
39  
40  /**
41   * A {@link TargetStrategy} that produces a {@link UserTarget}.
42   *
43   * @author Carl Harris
44   */
45  class UserTargetStrategy implements TargetStrategy {
46  
47    private static final USM usm = new USM(SecurityProtocols.getInstance(),
48        new OctetString(MPv3.createLocalEngineID()), 0);
49  
50    @Override
51    public Target newTarget(SnmpTarget target) {
52      if (!(target instanceof SnmpV3Target)) return null;
53      SnmpV3Target v3Target = (SnmpV3Target) target;
54      Assert.notNull(v3Target.getSecurityName(), "securityName is required");
55      if (v3Target.getAuthPassphrase() != null) {
56        Assert.notNull(v3Target.getAuthType(), "authType is required");
57      }
58      if (v3Target.getPrivPassphrase() != null){
59        Assert.notNull(v3Target.getPrivType(), "privType is required");
60      }
61      
62      OctetString securityName = new OctetString(v3Target.getSecurityName());
63      
64      usm.addUser(new UsmUser(securityName, 
65              authType(v3Target), new OctetString(v3Target.getAuthPassphrase()),
66              privType(v3Target), new OctetString(v3Target.getPrivPassphrase())));    
67  
68      UserTarget userTarget = new UserTarget();
69      userTarget.setSecurityName(securityName);
70      userTarget.setVersion(SnmpConstants.version3);
71  
72      if (v3Target.getAuthType() == null && v3Target.getPrivType() == null) {
73        userTarget.setSecurityLevel(SecurityLevel.NOAUTH_NOPRIV);
74      }
75      else if (v3Target.getPrivType() == null) {
76        userTarget.setSecurityLevel(SecurityLevel.AUTH_NOPRIV);
77      }
78      else if (v3Target.getAuthType() != null) {
79        userTarget.setSecurityLevel(SecurityLevel.AUTH_PRIV);
80      }
81      else {
82        throw new IllegalStateException(
83            "cannot support privacy without authentication");
84      }
85  
86      return userTarget;
87    }
88  
89    private OID authType(SnmpV3Target v3Target) {
90      switch (v3Target.getAuthType()) {
91        case SHA:
92          return AuthSHA.ID;
93        case MD5:
94          return AuthMD5.ID;
95        default:
96          throw new IllegalArgumentException("unrecognized auth type");
97      }
98    }
99    
100   private OID privType(SnmpV3Target v3Target) {
101     switch (v3Target.getPrivType()) {
102       case DES:
103         return PrivDES.ID;
104       case DES3:
105         return Priv3DES.ID;
106       case AES128:
107         return PrivAES128.ID;
108       case AES192:
109         return PrivAES192.ID;
110       case AES256:
111         return PrivAES256.ID;
112       default:
113         throw new IllegalArgumentException("unrecognized privacy type");
114     }
115   }
116 
117 }
