18  package org.soulwing.snmp.provider.mibble;
19  
20  import org.soulwing.snmp.Formatter;
21  
22  import net.percederberg.mibble.MibType;
23  import net.percederberg.mibble.MibTypeSymbol;
24  import net.percederberg.mibble.MibTypeTag;
25  import net.percederberg.mibble.MibValueSymbol;
26  import net.percederberg.mibble.snmp.SnmpObjectType;
27  import net.percederberg.mibble.snmp.SnmpTextualConvention;
28  import net.percederberg.mibble.type.BitSetType;
29  import net.percederberg.mibble.type.ChoiceType;
30  import net.percederberg.mibble.type.IntegerType;
31  
32  class ObjectFormatter implements Formatter {
33  
34    // Pre-SMIv2 textual conventions and types
35    private static final String NETWORK_ADDRESS_TYPE = "NetworkAddress";  // RFC 1155 section 3.2.3.1
36    private static final String DISPLAY_STRING_TYPE = "DisplayString";    // RFC 1213 section 3.2
37    
38    // default display hints
39    private static final String INTEGER_HINT = "d";
40    private static final String DISPLAY_STRING_HINT = "1a";
41    private static final String OCTET_STRING_HINT = "1x:";
42    private static final String IP_ADDRESS_HINT = "1d.";
43    
44    private final Formatter delegate;
45  
46    public ObjectFormatter(MibValueSymbol symbol) {
47      if (!(symbol.getType() instanceof SnmpObjectType)) {
48        this.delegate = new ToStringFormatter();
49      }
50      else {
51        try {
52          this.delegate = formatter(((SnmpObjectType) symbol.getType()).getSyntax());
53        }
54        catch (IllegalArgumentException ex) {
55          throw new IllegalArgumentException("cannot format symbol: " + symbol, ex);
56        }
57      }
58    }
59  
60    private static Formatter formatter(MibType syntax) {
61      if (syntax.hasTag(MibTypeTag.INTEGER)) {
62        if (((IntegerType) syntax).hasSymbols()) {
63          return new EnumFormatter(((IntegerType) syntax).getAllSymbols());
64        }
65        return new IntegerFormatter(displayHint(syntax, INTEGER_HINT));
66      }
67      else if (syntax.hasTag(MibTypeTag.BIT_STRING)) {
68        return new BitsFormatter(((BitSetType) syntax).getAllSymbols()); 
69      }
70      else if (syntax.hasTag(MibTypeTag.OCTET_STRING)) {
71        return new OctetStringFormatter(displayHint(syntax, OCTET_STRING_HINT));
72      }
73      else if (syntax.hasTag(MibTypeTag.OBJECT_IDENTIFIER)) {
74        return new ObjectIdentifierFormatter();
75      }
76      else if (syntax.hasTag(MibTypeTag.APPLICATION_CATEGORY, SnmpApplicationTypes.IP_ADDRESS)) {
77        return new OctetStringFormatter(IP_ADDRESS_HINT);
78      }
79      else if (syntax.hasTag(MibTypeTag.APPLICATION_CATEGORY, SnmpApplicationTypes.GAUGE32)) {
80        return new IntegerFormatter(displayHint(syntax, INTEGER_HINT));
81      }
82      else if (syntax.hasTag(MibTypeTag.APPLICATION_CATEGORY, SnmpApplicationTypes.TIME_TICKS)) {
83        return new TimeTicksFormatter(displayHint(syntax, null));
84      }
85      else if (syntax.hasTag(MibTypeTag.APPLICATION_CATEGORY, SnmpApplicationTypes.OPAQUE)) {
86        return new OctetStringFormatter(OCTET_STRING_HINT);
87      }
88      else if (syntax.hasTag(MibTypeTag.APPLICATION_CATEGORY, SnmpApplicationTypes.COUNTER32)
89          || syntax.hasTag(MibTypeTag.APPLICATION_CATEGORY, SnmpApplicationTypes.COUNTER64)) {
90        return new IntegerFormatter(INTEGER_HINT);
91      }
92      else if (syntax instanceof ChoiceType) {
93        if (syntax.getReferenceSymbol() != null) {
94          if (syntax.getReferenceSymbol().getName().equals(NETWORK_ADDRESS_TYPE)) {
95            return new OctetStringFormatter(IP_ADDRESS_HINT);
96          }
97        }
98      }
99      throw new IllegalArgumentException("unknown syntax: " + syntax);
100   }
101   
102   private static String displayHint(MibType syntax, String defaultHint) {
103     SnmpTextualConvention tc = textualConvention(syntax);
104     if (tc != null) {
105       if (tc.getDisplayHint() != null) {
106         return tc.getDisplayHint();
107       }
108     }
109     if (syntax.getReferenceSymbol() != null) {
110       if (syntax.getReferenceSymbol().getName().equals(DISPLAY_STRING_TYPE)) {
111         return DISPLAY_STRING_HINT;
112       }
113     }
114     return defaultHint;
115   }
116   
117   private static SnmpTextualConvention textualConvention(MibType syntax) {
118     SnmpTextualConvention tc = null;
119     if (syntax.getReferenceSymbol() != null) {
120       if (syntax.getReferenceSymbol().getType() instanceof SnmpTextualConvention) {
121         tc = (SnmpTextualConvention) syntax.getReferenceSymbol().getType();
122       }
123     }
124     return tc;
125   }
126 
127   public String format(Object value) {
128     return delegate.format(value);
129   }
130 
131 }
