19  package org.soulwing.snmp;
20  
21  /**
22   * A component that listens for SNMP notifications (traps, informs) and
23   * makes them available to registered handlers.
24   * <p>
25   * Conceptually, a listener represents a provider component that listens for
26   * incoming SNMP notifications at the transport layer of the network.  Typically,
27   * an implementation binds a socket to an IP address and port and receives all
28   * incoming SNMP notifications received on that socket.
29   * <p>
30   * A listener maintains an ordered collection of handlers, based on a priority
31   * specified when each handler is registered.  When an SNMP notification is
32   * received, registered handlers are invoked in priority order.  When a handler
33   * indicates that it handled the event (via the return value of
34   * {@link SnmpNotificationHandler#handleNotification(SnmpNotificationEvent)}),
35   * handlers of lower priority are not invoked.
36   * <p>
37   * The listener can be used to implement a strategy for handling different types
38   * of notifications, where a given handler supports one notification type (or
39   * perhaps a set of closely related types).  For each received notification,
40   * the listener will ask each registered handler (in priority order) to handle
41   * the notification, stopping with the first handler that claims to have done so.
42   *
43   * @author Carl Harris
44   */
45  public interface SnmpListener {
46  
47    /**
48     * Adds a notification handler.
49     * <p>
50     * The handler will be added to the collection such that it is notified
51     * after all previously registered handlers.
52     * <p>
53     * This method has no effect if a handler identical to {@code handler} is
54     * already registered.
55     * @param handler the handler to add
56     */
57    void addHandler(SnmpNotificationHandler handler);
58  
59    /**
60     * Adds a notification handler.
61     * <p>
62     * The handler will be added to the collection, such that it is notified
63     * after all previously registered handlers whose priority is less than or
64     * equal to the specified {@code priority}.
65     * <p>
66     * This method has no effect if a handler identical to {@code handler} is
67     * already registered.
68     * @param handler the handler to add
69     * @param priority priority for the handler
70     */
71    void addHandler(SnmpNotificationHandler handler, int priority);
72  
73    /**
74     * Removes a notification handler.
75     * <p>
76     * This method has no effect if {@code handler} is not registered
77     * @param handler the handler to remove
78     */
79    void removeHandler(SnmpNotificationHandler handler);
80  
81    /**
82     * Closes this listener.
83     * <p>
84     * After the listener has been closed, no further notifications will be
85     * distributed to handlers.
86     */
87    void close();
88  
89  }
