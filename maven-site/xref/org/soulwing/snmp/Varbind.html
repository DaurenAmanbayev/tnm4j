18  package org.soulwing.snmp;
19  
20  public interface Varbind extends ObjectValue {
21  
22    /**
23     * Gets the MIB name of the bound object.
24     * @return name
25     */
26    String getName();
27    
28    /**
29     * Gets the object identifier of the named object as a dotted decimal string.
30     * @return OID string
31     */
32    String getOid();
33    
34    /**
35     * Gets the bindings that represent the index values the bound value.
36     * @return if the bound value is a member of a conceptual table, the
37     *    variable bindings representing the object's table index are returned;
38     *    an empty array is returned for an object that is not a table member
39     */
40    Varbind[] getIndexes();
41    
42  }
