18  package org.soulwing.snmp;
19  
20  import java.io.Closeable;
21  
22  
23  /**
24   * A context for performing SNMP operations with an SNMP agent.
25   * <p>
26   * A context is a lightweight object that represents a single SNMP agent
27   * on a network.  Implementations may be designed such that a single 
28   * provider object is shared by many context objects.
29   * <p>
30   * A context is obtained from an {@link SnmpFactory}.  When the context
31   * is no longer used, the {@link SnmpContext#close()} method should be
32   * called to release any provider resources associated with the context.
33   *
34   * @author Carl Harris
35   */
36  public interface SnmpContext extends SnmpOperationFactory,
37      SnmpOperations, SnmpAsyncOperations, Closeable {
38  
39    /**
40     * Gets the target of this context.
41     * @return target
42     */
43    SnmpTarget getTarget();
44    
45    /**
46     * Gets the MIB associated with this context.
47     * @return MIB
48     */
49    Mib getMib();
50  
51    /**
52     * Closes this context, releasing any provider resources associated
53     * with it.
54     */
55    void close();
56    
57    /**
58     * Constructs a new {@link Varbind} suitable for use in operations
59     * performed via this context.
60     * @param oid MIB name or dotted-decimal OID
61     * @param value value whose type is appropriate for the specified object
62     *    identifier
63     * @return bound instance of the given value
64     */
65    Varbind newVarbind(String oid, Object value);
66    
67  }
