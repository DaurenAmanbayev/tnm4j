19  package org.soulwing.snmp;
20  
21  /**
22   * An object that walks a conceptual table asynchronously.
23   * <p>
24   * An asynchronous walker is an operation that when invoked, provides
25   * a reference to a walker that can be used to retrieve rows from the
26   * table.  It's {@link #next()} method will throw a {@link WouldBlockException}
27   * if an additional operation is required to get the next row (or to
28   * determine that the end of table has been reached).  On receipt of this
29   * exception, the caller should then invoke the walker (as an operation) to
30   * continue.
31   *
32   * @author Carl Harris
33   */
34  public interface SnmpAsyncWalker<V> extends SnmpWalker<V>, 
35      SnmpOperation<SnmpAsyncWalker<V>> {
36  
37    /**
38     * Gets the next response from this walker.
39     * @return next response or {@code null} to indicate that the end of the
40     *    subject table has been reached. 
41     * @throws WouldBlockException if another response is not available because
42     *    it has not yet been retrieved from the remote agent
43     */
44    SnmpResponse<V> next() throws WouldBlockException;
45    
46  }
