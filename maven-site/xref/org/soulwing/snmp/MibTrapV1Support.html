19  package org.soulwing.snmp;
20  
21  /**
22   * A provider of specialized MIB support for handling SNMPv1 TRAP PDUs.
23   * <p>
24   * SNMPv1 traps contain several fields that are represent in a manner similar
25   * to variable bindings, but for which no managed object type is defined.  An
26   * implementation of this class provides basic formatting support for these
27   * fields.
28   */
29  public interface MibTrapV1Support {
30  
31    /**
32     * Gets a formatter for the {@code generic-trap} field of a V1 TRAP PDU.
33     * @return formatter
34     */
35    Formatter getGenericTrapFormatter();
36  
37    /**
38     * Gets a formatter for the {@code specific-trap} field of a V1 TRAP PDU.
39     * @return formatter
40     */
41    Formatter getSpecificTrapFormatter();
42  
43    /**
44     * Gets a formatter for the {@code timestamp} field of a V1 TRAP PDU.
45     * @return formatter
46     */
47    Formatter getTimestampFormatter();
48  
49  }
