18  package org.soulwing.snmp;
19  
20  
21  public class SnmpV2cWalkDemo {
22  
23    
24    public static void main(String[] args) throws Exception {    
25      final String ipAddress = System.getProperty("agent.ip", "10.0.0.1");
26      final String community = System.getProperty("agent.community", "public");
27  
28      SimpleSnmpV2cTarget target = new SimpleSnmpV2cTarget();
29      target.setAddress(ipAddress);
30      target.setCommunity(community);
31      
32      Mib mib = MibFactory.getInstance().newMib();
33      mib.load("RFC1213-MIB");
34      mib.load("IF-MIB");
35          
36      SnmpContext snmp = SnmpFactory.getInstance().newContext(target, mib);
37  
38      SnmpWalker<VarbindCollection> walker = snmp.walk(
39          1, "sysUpTime", "ifIndex", "ifName", "ifInOctets", "ifHCInOctets");
40      
41      VarbindCollection row = walker.next().get();
42      while (row != null) { 
43        System.out.format("%s %d %s %s %s\n", 
44            row.get("sysUpTime"), 
45            row.get("ifIndex").asInt(),
46            row.get("ifName"),
47            row.get("ifInOctets"), 
48            row.get("ifHCInOctets"));
49        row = walker.next().get();
50      }
51  
52      snmp.close();
53      SnmpFactory.getInstance().close();
54    }
55  
56  }
