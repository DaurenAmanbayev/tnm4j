18  package org.soulwing.snmp;
19  
20  import java.util.regex.Matcher;
21  import java.util.regex.Pattern;
22  
23  public class SnmpV2cDemo {
24  
25    private static final String NOT_AVAILABLE = "N/A";
26    
27    private static final Pattern CISCOIOS_PATTERN =
28        Pattern.compile("Software \\(([^)]+)\\).*Version  *([^ ,]+) *");
29    
30    public static void main(String[] args) throws Exception {    
31      final String deviceName = System.getProperty("agent.name", "foo");
32      final String ipAddress = System.getProperty("agent.ip", "10.0.0.1");
33      final String community = System.getProperty("agent.community", "public");
34  
35      SimpleSnmpV2cTarget target = new SimpleSnmpV2cTarget();
36      target.setAddress(ipAddress);
37      target.setCommunity(community);
38      
39      Mib mib = MibFactory.getInstance().newMib();
40      mib.load("RFC1213-MIB");
41          
42      SnmpContext snmp = SnmpFactory.getInstance().newContext(target, mib);
43  
44      VarbindCollection varbinds = snmp.getNext("sysDescr", "sysUpTime").get();
45      String sysDescr = varbinds.get("sysDescr").asString();
46      String sysUpTime = varbinds.get("sysUpTime").asString();
47      Matcher matcher = CISCOIOS_PATTERN.matcher(sysDescr);
48      boolean found = matcher.find();
49      String software = found ? matcher.group(1) : NOT_AVAILABLE;
50      String version = found ? matcher.group(2) : NOT_AVAILABLE;
51      
52      System.out.format("%-12s %-15s %-24s %-24s %s\n", 
53          "Device Name", "IP Address", "Software", "Version" , "Up Time");
54      
55      System.out.format("%-12s %-15s %-24s %-24s %s\n",
56          deviceName, ipAddress, software, version, sysUpTime);
57  
58      SnmpFactory.getInstance().close();
59    }
60    
61  }
